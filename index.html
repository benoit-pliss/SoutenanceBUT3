						FROM od_order_detail od
							INNER JOIN od_orders o ON (o.id_order = od.id_order)
							AND o.invoice_date BETWEEN :date_debut AND :date_fin
							LEFT JOIN od_order_grossiste_data ogd ON (o.id_order = ogd.id_order)
							AND ogd.id_order IS NULL
							INNER JOIN erp_produit_prod_presta p ON (od.product_id = p.id_product
							AND od.product_attribute_id = p.id_product_attribute)
							INNER JOIN erp_produit_prod pp ON (p.id_produit_prod = pp.id_produit_prod)
						WHERE o.invoice_number != 0
						AND od.product_id IN (" . implode(',', $this->liste_id_product_produit_prod) . ")
						GROUP BY od.product_id, od.product_attribute_id;
